% (command_citationfile)
\newcommand{\citationfile}[1]{
	\bibliography{#1}
}

% (command_citationstyle)
\newcommand{\citationstyle}[1]{
	\usepackage[natbib=true,style=#1,maxnames=99]{biblatex}
}

% (command_footer)
\newcommand{\footer}{
	\rfoot{\small\scshape\thepage\hspace{2pt} of \thelastpage}
}

% (command_fontstyle)
\newcommand{\fontstyle}[1]{
	\setmainfont{#1}
}

% (command_header)
\newcommand{\header}[3]{
	\lhead{\small #1 -- #2}
	\rhead{\small\scshape #3}
}

% (command_linespacing)
\newcommand{\linespacing}[1]{
	\setstretch{#1}
}

% (command_thelastpage)
\newcommand{\thelastpage}{
	\pageref{LastPage}
}

% (command_referencelist)
\newcommand{\referencelist}{
	\printbibliography[title={\normalfont\selectfont\scshape\large References}]
}
